    private void printTable(JTable tabela, Tabela table) {

        tabela.setDefaultRenderer(Object.class, new DefaultTableCellRenderer() {

            @Override
            public Component getTableCellRendererComponent(JTable tabela, Object value, boolean isSelected, boolean hasFocus, int row, int column) {
                //  JLabel label = (JLabel) super.getTableCellRendererComponent(table, value, true, hasFocus, linha, column);
                JLabel label1 = (JLabel) super.getTableCellRendererComponent(tabela, value, false, hasFocus, row, column);

                Color c = Color.WHITE;

                Object texto = tabela.getValueAt(row, 1);

                label1.setBackground(c);
                // tabela.setSelectionForeground(Color.green);
                return label1;
            }

        });

        table.fireTableDataChanged();

        jClock.setText("Clock: " + (escalonator.getTempoAtual() - 1));
        jQuantum.setText("Quantum Restante:" + escalonator.getXeon().getQuantumRestante());

        if (!escalonator.getXeon().isVazio()) {
            jProcessando.setText(escalonator.getXeon().getExecutando().getTempoChegada() + "   " + escalonator.getXeon().getExecutando().getTempoServico() + "   " + escalonator.getXeon().getExecutando().getPrioridadeOriginal() + "   " + escalonator.getXeon().getExecutando().getRestante());
        } else {
            jProcessando.setText("Processador vazio");
        }

        if (escalonator.getProcessosAtivos().isEmpty() && escalonator.getListaDeProcessos().isEmpty() && escalonator.getXeon().isVazio()) {
            JOptionPane.showMessageDialog(rootPane, "Finalizou simulação");
        }

    }
